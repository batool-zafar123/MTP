### 2.1. Shared Design System & Theme Styling
All three pages (`frontend.html`, `templates.html`, `about.html`) must use the same fonts, layout widths, colors, and transition effects. To ensure consistency, you must support **Dark and Light Theme Toggling** using root CSS variables and persist it via `localStorage`.

Define the following CSS root variables in the head of each file or in a shared stylesheet. They represent the core design system:
```css
:root {
  /* Dark mode (default) colors matching the current frontend.html */
  --bg:          #0d0f12;
  --surface:     #13161b;
  --surface2:    #1a1e25;
  --border:      #252a34;
  --accent:      #c8f04e;   /* lime green */
  --accent2:     #4ef0c8;  /* teal */
  --text:        #e8eaf0;
  --text-muted:  #6b7280;
  --danger:      #f04e6b;
  --font-ui:     'Syne', sans-serif;
  --font-mono:   'DM Mono', monospace;
  --font-body:   'DM Sans', sans-serif;
  --bar-h:       56px;
  --divider-w:   5px;
}

body.light-theme {
  /* Light mode overrides */
  --bg:          #f4f6f9;
  --surface:     #ffffff;
  --surface2:    #eaeef3;
  --border:      #d1d5db;
  --accent:      #4f46e5;   /* indigo */
  --accent2:     #0d9488;  /* teal */
  --text:        #111827;
  --text-muted:  #4b5563;
  --danger:      #e11d48;
}
```

#### Theme Persistence Logic:
Add this script directly inside the `<head>` tag of all three HTML files to prevent flash-of-dark-theme issues:
```javascript
(function() {
  const savedTheme = localStorage.getItem('portify-theme');
  if (savedTheme === 'light') {
    document.documentElement.classList.add('light-theme');
  }
})();
```
In the body, add a utility class toggle function on a click event for the toggle button:
```javascript
const themeToggleBtn = document.getElementById('theme-toggle');
themeToggleBtn.addEventListener('click', () => {
  document.body.classList.toggle('light-theme');
  const isLight = document.body.classList.contains('light-theme');
  localStorage.setItem('portify-theme', isLight ? 'light' : 'dark');
  // Sync across opened documents
  if (isLight) {
    document.documentElement.classList.add('light-theme');
  } else {
    document.documentElement.classList.remove('light-theme');
  }
});
```

<script>
  // Dynamic selection indicator and update mechanisms
  function selectTemplate(id) {
    localStorage.setItem('selected_template', id);
    updateSelectButtons();
  }
 
  function updateSelectButtons() {
    const activeId = localStorage.getItem('selected_template') || '1';
    document.querySelectorAll('.btn-select-tmpl').forEach(btn => {
      const card = btn.closest('.template-card');
      const cardId = card.getAttribute('data-id');
      if (cardId === activeId) {
        btn.classList.add('active');
        btn.textContent = 'Selected';
      } else {
        btn.classList.remove('active');
        btn.textContent = 'Select Template';
      }
    });
  }
 
  // Initial active settings checking
  window.addEventListener('DOMContentLoaded', () => {
    updateSelectButtons();
    
    // Connect theme toggle button if clicked
    const themeToggleBtn = document.getElementById('theme-toggle');
    themeToggleBtn.addEventListener('click', () => {
      document.body.classList.toggle('light-theme');
      const isLight = document.body.classList.contains('light-theme');
      localStorage.setItem('portify-theme', isLight ? 'light' : 'dark');
      if (isLight) {
        document.documentElement.classList.add('light-theme');
      } else {
        document.documentElement.classList.remove('light-theme');
      }
    });
  });
</script>

## 4. PHASE 3: PYTHON SERVER REFACTORING (`server.py`)

Refactor `server.py` to:
1. Fix the typo serving `frontend.htm` to `frontend.html`.
2. Extract the `template` parameter from the JSON request payload.
3. Pass the template number as a shell argument when invoking the compiled `main.exe` executable.

```python
from flask import Flask, request, send_file, send_from_directory
from flask_cors import CORS
import subprocess
import os

app = Flask(__name__)
CORS(app)

@app.route('/')
def home():
    # FIXED TYPO: Serve frontend.html instead of non-existent frontend.htm
    return send_file('frontend/frontend.html')

@app.route('/output/<path:filename>')
def serve_output(filename):
    """Serve files from the output directory"""
    return send_from_directory('output', filename)

@app.route('/generate', methods=['POST'])
def handle_generation():
    data = request.json
    markdown_text = data.get('markdown', '')
    template_id = data.get('template', '1') # Fallback to template 1
    
    # Save markdown to input.md
    with open('input.md', 'w', encoding='utf-8') as file:
        file.write(markdown_text)
    
    # Run dynamic C++ program with dynamic template identifier
    try:
        # Check running OS context (add .exe only if on windows context)
        exe_path = './backend/main.exe' if os.name == 'nt' else './backend/main'
        
        # Pass template_id as a command line string parameter
        subprocess.run([exe_path, str(template_id)], check=True)
        
        return {"status": "success", "message": f"Portfolio generated using Template {template_id}!"}, 200
    except FileNotFoundError:
        return {"status": "error", "message": "C++ program executable not found. Make sure to compile backend/main.cpp first!"}, 500
    except subprocess.CalledProcessError:
        return {
            "status": "error", 
            "message": "C++ program is saying that the structure of md is disturbed. Try reloading..."
        }, 500
    except Exception as e:
        return {"status": "error", "message": str(e)}, 500

if __name__ == '__main__':
    app.run(port=8000, debug=False)
```
